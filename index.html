let isDragging = false; let offsetX = 0; let offsetY = 0; const box = document.getElementById("YoudaoBox"); // ❗安全检查（避免 null 报错） if (box) { box.addEventListener("mousedown", function (e) { isDragging = true; offsetX = e.clientX - box.offsetLeft; offsetY = e.clientY - box.offsetTop; }); } // ❗拖动 document.addEventListener("mousemove", function (e) { if (!isDragging || !box) return; box.style.left = (e.clientX - offsetX) + "px"; box.style.top = (e.clientY - offsetY) + "px"; box.style.right = "auto"; }); // ❗结束拖动 document.addEventListener("mouseup", function () { isDragging = false; }); // ❗点击空白关闭 document.addEventListener("click", function (e) { const popup = document.getElementById("popup"); const box = document.getElementById("YoudaoBox"); const isWord = e.target.classList.contains("en"); const isInsidePopup = popup && popup.contains(e.target); const isInsideBox = box && box.contains(e.target); if (!isWord && !isInsidePopup && !isInsideBox) { if (popup) popup.style.display = "none"; if (box) box.style.display = "none"; } });
